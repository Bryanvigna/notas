	//*SINAIS OPERADORES*//
=	atribui algo
+	soma normal
-	subtração normal
*	multiplicação normal
/	divisão real(normal)
%	divisão inteira(resto da divisão)
**	Elevado ao quadrado

	//*OPERADORES LÓGICOS*//
!	negação
&&	conjunção(e)
||	disjunção(ou)
ordem de execução dos operadores:
1ª:!
2ª:&&
3ª:||

	//*RELACIONAIS*//
>	maior que
<	menor que
>=	maior ou igual
<=	menor ou igual
==	igual em grandeza
===	igual restrito(igual em number, string, etc.)
!=	diferente em grandeza
!==	diferente restrito(diferente em number, string, etc.)
(obs: respondidos em valor booleano (true/false))

	//*OPERADOR TERNÁRIO*//
?	se for verdadeiro aparece esse
:	se não for verdadeiro aparece esse
